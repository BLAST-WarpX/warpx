#################################
########## MY CONSTANTS #########
#################################
my_constants.mc2   = m_e*clight*clight
my_constants.nano  = 1.0e-9
my_constants.micro = 1.e-6
my_constants.GeV   = q_e*1.e9

# BEAMS
my_constants.beam_energy = 187.5*GeV
my_constants.beam_gamma  = beam_energy/mc2
my_constants.beam_npart  = 2.3e11
my_constants.nmacropart  = 1e4
my_constants.beam_charge = q_e * beam_npart
my_constants.sigmax      = 38200*nano
my_constants.sigmay      = 68.1*nano
my_constants.sigmaz      = 2540.*micro
my_constants.beam_uth    = 0.
my_constants.mux         = 0.
my_constants.muy         = 0.
my_constants.muz         = 4*sigmaz
my_constants.emitx       = 535.7*micro
my_constants.emity       = 1.064*micro
my_constants.dux         = emitx / sigmax
my_constants.duy         = emity / sigmay

# BOX
my_constants.Lx = 32*sigmax
my_constants.Ly = 128*sigmay
my_constants.Lz = 16*sigmaz
my_constants.nx = 128
my_constants.ny = 128
my_constants.nz = 200
my_constants.dx = Lx/nx
my_constants.dy = Ly/ny
my_constants.dz = Lz/nz

# TIME
my_constants.T = 0.5*Lz/clight
my_constants.dt = T / nz


my_constants.nt = floor(T/dt) - 1
my_constants.coll_time = abs(muz)/clight
my_constants.coll_step = floor(nt*coll_time/T)


warpx.used_inputs_file = warpx_used_inputs.txt


my_constants.sigma_bw_max = 1.7e-32 # m^2
my_constants.probability_estimate =   beam_npart / nmacropart * sigma_bw_max * 2 * clight * dt  / (dx * dy * dz)

my_constants.probability_target_value = 0.01
my_constants.multiplier =  probability_target_value / probability_estimate
my_constants.probability_threshold = 0.1
